I/O registers!


control pins -> read datasheet 3 registers, we use 2 pin B0, lowest bit pin 8 on Arduino more bits than registers! 8 in register, 6 on hardware 
#include <stdint.h>

// Memory mapped I/O (values from the datasheet).
#define DDRB  *(volatile uint8_t*)(0x24)
#define PORTB *(volatile uint8_t*)(0x25)

int main() {
    DDRB  |= 1 << 5; // set pin PB5 as an output
    PORTB |= 1 << 5; // set pin PB5 high

    // wait forever
    while (1) {
    }

    return 0; // unreachable (satisfy the compiler)
}
sbi 0x04, 0x5   ; set bit 5 in DDRB  (0x24)
sbi 0x05, 0x5   ; set bit 5 in PORTB (0x25)
rjmp .-2        ; jump back to the same instruction (loop)
same thing SBI: set a given bit in I/O register - leaves all others unaffected (0 through 7) special for AVR

void _start() {
    DDRB |= (1<<5); // set pin PB5 as an output

    // main loop
    while (1) {
        for (long i=1000000; i != 0; i--) {
            PORTB |= (1<<5); // set pin PB5 high (set bit)
        }
        for (long i=1000000; i != 0; i--) {
            PORTB &= ~(1<<5); // set pin PB5 low (clear bit)
        }
    }
}
